Function as a parameter

integrate <- function(f, a, b) {
  n <- 10
  w <- (b - a) / n
  sum <- 0.0
  for (i in 1:n) {
    xi <- a + (i - 1) * w
    sum <- sum + w * f(xi)
  }
  sum
}

linear <- function(x) {
  x
}

area <- integrate(linear, 0, 10)
print(area)

Returning a Function

operation <- function(op) {

  add <- function(a, b) {
    a + b
  }

  sub <- function(a, b) {
    a - b
  }

  if (op == "add") {
    add
  } else if (op == "sub") {
    sub
  }

}

add <- operation("add")
sub <- operation("sub")
none <- operation("none")

add(2, 1)
sub(2, 1)
print(none)
